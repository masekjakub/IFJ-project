GRAMATIKA

<prog> => BEGIN DECLARE_ST <stat_list>

<id> => ID
<id> => FUNID(<args>)

<stat_list> => EPILOG EOF
<stat_list> => EOF
<stat_list> => <stat> <stat_list>
<stat> => IF (<expr>) {<stat_list>} else {<stat_list>}
<stat> => WHILE (<expr>) {<stat_list>}
<stat> => FUNCTION FUNID (<params>) <funcdef>
<stat> => <assign>

<funcdef> => {<stat_list>}
<funcdef> => : TYPE {<stat_list>}

<assign> => ID = <expr> ;
<assign> => ID <expr> ;
<assign> => <expr> ;

<param> => TYPE ID
<param> => ? TYPE ID
<params> => <param> <params_2> 
<params> => epsilon
<params_2> => , <param> <params_2> 
<params_2> => epsilon

<args> => epsilon
<args> => <id> <args_2>
<args_2> => , <id> <args_2>
<args_2> => epsilon


<return> => RETURN <expr> ;
